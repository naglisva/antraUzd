79template<typename T>
80double mediana2(T vekt)
81{
82 size_t size = vekt.size();
83
84 if (size == 0)
85 {
86 return 0;
87 }
88 else
89 {
90 if constexpr (is_same_v<T, vector<typename T::value_type>>)
91 {
92 sort(vekt.begin(), vekt.end());
93 }
94 else if constexpr (is_same_v<T, list<typename T::value_type>>)
95 {
96 vekt.sort();
97 }
98
99 if (size % 2 == 0)
100 {
101 auto it = next(vekt.begin(), size / 2 - 1);
102 return static_cast<double>(*it + *(++it)) / 2;
103 }
104 else
105 {
106 return *next(vekt.begin(), size / 2);
107 }
108 }
109}
